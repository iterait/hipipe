373  template<typename Rng>
374  auto unzip_impl(Rng& range_of_tuples)
375  {
376  using tuple_type = ranges::range_value_type_t<Rng>;
377  constexpr auto tuple_size = std::tuple_size<tuple_type>{};
378  constexpr auto indices = std::make_index_sequence<tuple_size>{};
379  std::size_t reserve_size = detail::safe_reserve_size(range_of_tuples);
380 
381  auto tuple_of_ranges = detail::vectorize_tuple<tuple_type>(indices);
382  utility::tuple_for_each(
383  tuple_of_ranges, [reserve_size](auto& rng) { rng.reserve(reserve_size); });
384 
385  for (auto& v : range_of_tuples) {
386  detail::push_unzipped(tuple_of_ranges, std::move(v), indices);
387  }
388 
389  return tuple_of_ranges;
390  }
