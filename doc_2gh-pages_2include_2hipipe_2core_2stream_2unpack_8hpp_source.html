26  // replace the columns in the stream with their values (i.e., raw batches)
27  template<typename... FromColumns>
28  constexpr auto unpack_columns()
29  {
30  return ranges::view::transform([](auto source) {
31  auto proj = [](auto& column) { return std::move(column.value()); };
32  auto subtuple = utility::tuple_type_view<FromColumns...>(source);
33  return utility::tuple_transform(std::move(subtuple), std::move(proj));
34  });
35  }

37  template<std::size_t Dim, bool OnlyOne, typename... FromColumns>
38  struct unpack_impl
39  {
40  template<typename Rng>
41  static constexpr auto impl(Rng range_of_tuples)
42  {
43  // project the selected columns to their values (i.e., batches)
44  auto raw_range_of_tuples = range_of_tuples | unpack_columns<FromColumns...>();
45  auto tuple_of_batches = utility::unzip(std::move(raw_range_of_tuples));
46  // flatten the values in each column upto the given dimension
47  return utility::tuple_transform(std::move(tuple_of_batches), [](auto&& batch_range) {
48  // make sure to convert the flat view to std::vector to avoid dangling ref
49  return utility::flat_view<Dim+1>(batch_range)
50  | ranges::view::move | ranges::to_vector;
51  });
52  }
53  };

55  // if only a single column is unpacked, do not return a tuple
56  template<std::size_t Dim, typename FromColumn>
57  struct unpack_impl<Dim, true, FromColumn>
58  {
59  template<typename Rng>
60  static constexpr auto impl(Rng&& range_of_tuples)
61  {
62  return std::get<0>(
