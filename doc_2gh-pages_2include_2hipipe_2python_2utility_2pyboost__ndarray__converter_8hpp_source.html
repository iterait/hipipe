107 template<typename T>
108 PyObject* to_ndarray(const std::vector<T>& vec)
109 {
110  auto data = std::make_unique<detail::ndarray_type_t<T>[]>(vec.size());
111  for (std::size_t i = 0; i < vec.size(); ++i) data[i] = detail::to_ndarray_element(vec[i]);
112  npy_intp dims[1]{static_cast<npy_intp>(vec.size())};
113 
114  PyObject* arr = PyArray_SimpleNewFromData(
115  1, dims, detail::to_ndarray_typenum<T>(), reinterpret_cast<void*>(data.release()));
116  if (!arr) throw std::runtime_error{"Cannot create Python NumPy ndarray."};
117  // we have to tell NumPy to delete the data when the array is removed
118  PyArray_ENABLEFLAGS(reinterpret_cast<PyArrayObject*>(arr), NPY_ARRAY_OWNDATA);
119  return arr;
120 }
