113  std::size_t insert(T val)
114  {
115  if (contains(val)) {
116  throw std::invalid_argument{"The element is already present in the index mapper."};
117  }
118  val2idx_[val] = idx2val_.size();
119  idx2val_.push_back(std::move(val));
120  return idx2val_.size() - 1;
