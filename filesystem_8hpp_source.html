24  // Generate random path inside the temp directory. This path is not
25  // checked for existence.
26  std::experimental::filesystem::path generate_temp_path(std::string pattern)
27  {
28  constexpr const char* hexchars = "0123456789abcdef";
29  static thread_local std::mt19937 gen{std::random_device{}()};
30  std::uniform_int_distribution<> dis{0, 15};
31  for (char& ch: pattern) {
32  if (ch == '%') ch = hexchars[dis(gen)];
33  }
34  namespace fs = std::experimental::filesystem;
35  return fs::temp_directory_path() / pattern;
36  }

45 std::experimental::filesystem::path create_temp_directory(const std::string &pattern)
46 {
47  namespace fs = std::experimental::filesystem;

49  while (max_retries--) {
