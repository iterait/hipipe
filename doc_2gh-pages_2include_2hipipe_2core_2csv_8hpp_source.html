105  // parse csv field and return whether the next separator is found
106  std::tuple<std::string, bool> parse_field()
107  {
108  std::string field;
109  char c;
110  while (in_->get(c)) {
111  if (c == separator_) {
112  return {std::move(field), true};
113  } else if (c == '\n') {
114  return {std::move(field), false};
115  }
116  field.push_back(c);
117  }
118  return {std::move(field), false};
119  }
120 
121  // parse csv row
122  void next()
123  {
124  if (!in_->good() || row_position_ == RowPosition::Last) {
125  row_position_ = RowPosition::End;
126  return;
127  }
128 
129  // temporarily set badbit exception mask
130  auto orig_exceptions = in_->exceptions();
131  in_->exceptions(orig_exceptions | std::istream::badbit);
132 
133  row_.clear();
134  bool has_next = true;
135  while (has_next && *in_ >> detail::blanks) {
136  std::string field;
137  // process quoted fields
138  if (in_->peek() == quote_) {
139  *in_ >> std::quoted(field, quote_, escape_);
140  if (in_->fail()) throw std::ios_base::failure{"Error while reading CSV field."};
141  std::tie(std::ignore, has_next) = parse_field();
142  }
143  // process unquoted fields
144  else {
145  std::tie(field, has_next) = parse_field();
146  boost::trim(field);
147  }
148  row_.push_back(std::move(field));
149  }
150 
151  // detect whether end of file is reached
152  *in_ >> std::ws;
153  in_->peek();
154  if (!in_->good()) {
155  row_position_ = RowPosition::Last;
156  }
157 
158  // reset exception mask
159  in_->exceptions(orig_exceptions);
160  }

194 inline dataframe<> read_csv(std::istream& in,
195  int drop = 0,
196  bool has_header = true,
197  char separator = ',',
198  char quote = '"',
199  char escape = '\\')
200 {
201  // header
202  std::vector<std::string> header;
203  // data
204  std::vector<std::vector<std::string>> data;
205  // load csv line by line
206  auto csv_rows =
207  csv_istream_range(in, separator, quote, escape)
208  | ranges::view::drop(drop)
209  | ranges::view::move;
210  auto csv_row_it = ranges::begin(csv_rows);
211  // load header if requested
212  std::size_t n_cols = -1;
213  if (has_header) {
214  if (csv_row_it == ranges::end(csv_rows)) {
215  throw std::ios_base::failure{"There has to be at least the header row."};
216  }
217  std::vector<std::string> csv_row = *csv_row_it;
218  n_cols = ranges::size(csv_row);
219  header = std::move(csv_row);
220  data.resize(n_cols);
221  ++csv_row_it;
222  }
223  // load data
224  for (std::size_t i = 0; csv_row_it != ranges::end(csv_rows); ++csv_row_it, ++i) {
225  std::vector<std::string> csv_row = *csv_row_it;
226  // sanity check row size
227  if (i == 0) {
228  if (has_header) {
229  if (ranges::size(csv_row) != n_cols) {
230  throw std::ios_base::failure{"The first row must have the same "
231  "length as the header."};
232  }
233  } else {
234  n_cols = ranges::size(csv_row);
235  data.resize(n_cols);
236  }
237  } else {
238  if (ranges::size(csv_row) != n_cols) {
239  throw std::ios_base::failure{"Row " + std::to_string(i)
240  + " has a different length "
241  + "(has: " + std::to_string(ranges::size(csv_row))
242  + " , expected: " + std::to_string(n_cols)
243  + ")."};
244  }
245  }
246  // store columns
247  for (std::size_t j = 0; j < ranges::size(csv_row); ++j) {
248  data[j].push_back(std::move(csv_row[j]));
249  }
250  }
251  return {std::move(data), std::move(header)};
252 }

287 template <typename Row>
288 std::ostream& write_csv_row(std::ostream& out,
289  Row&& row,
290  char separator = ',',
291  char quote = '"',
292  char escape = '\\')
293 {
294  // temporarily set badbit exception mask
295  auto orig_exceptions = out.exceptions();
296  out.exceptions(orig_exceptions | std::ostream::badbit);
297 
298  for (std::size_t i = 0; i < ranges::size(row); ++i) {
299  auto& field = row[i];
300  // output quoted string if it contains separator, double quote, newline or
301  // starts or ends with a whitespace
302  if (ranges::find_first_of(field, {separator, quote, '\n'}) != ranges::end(field)
303  || detail::trimmable(field)) {
304  out << std::quoted(field, quote, escape);
305  } else {
306  out << field;
307  }
308 
309  // output separator or newline
310  if (i + 1 < ranges::size(row)) out << separator;
311  else out << '\n';
312  }
313 
314  out.exceptions(orig_exceptions);
315  return out;
316 }
