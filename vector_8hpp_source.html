239 template<long NDims, typename Rng, typename ValT = ndim_type_t<Rng, NDims>>
240 Rng& ndim_resize(Rng& vec,
241  const std::vector<std::vector<long>>& vec_size,
242  ValT val = ValT{})
243 {
244  // check that the size is valid
245  assert(vec_size.size() == NDims);
246  static_assert(NDims <= ndims<Rng>{} - ndims<ValT>{});
247  for (std::size_t i = 1; i < vec_size.size(); ++i) {
248  assert(vec_size[i].size() == ranges::accumulate(vec_size[i-1], 0UL));
249  }
250  // build initial indices
251  std::vector<long> vec_size_idx(vec_size.size());
252  // recursively resize
253  detail::ndim_resize_impl<1, NDims>::impl(vec, vec_size, vec_size_idx, val);
254  return vec;
255 }

453  template<long N>
454  struct reshaped_view_impl_go {
455  static auto impl(const std::shared_ptr<std::vector<long>>& shape_ptr)
456  {
457  return ranges::view::chunk((*shape_ptr)[N-2])
458  | ranges::view::transform([shape_ptr](auto subview) {
459  return std::move(subview) | reshaped_view_impl_go<N-1>::impl(shape_ptr);
460  });
461  }
462  };
463 
464  template<>
465  struct reshaped_view_impl_go<1> {
466  static auto impl(const std::shared_ptr<std::vector<long>>&)
467  {
468  return ranges::view::all;
469  }
470  };
471 
472  template<long N, typename Rng>
473  auto reshaped_view_impl(Rng& vec, std::vector<long> shape)
474  {
475  assert(shape.size() == N);
476  auto flat = flat_view(vec);
477 
478  // if -1 present in the shape list, deduce the dimension
479  auto deduced_pos = ranges::find(shape, -1);
480  if (deduced_pos != shape.end()) {
481  auto flat_size = ranges::distance(flat);
482  auto shape_prod = -ranges::accumulate(shape, 1, std::multiplies<>{});
483  assert(flat_size % shape_prod == 0);
484  *deduced_pos = flat_size / shape_prod;
485  }
486 
487  // check that all the requested dimenstions have positive size
488  assert(ranges::all_of(shape, [](long s) { return s > 0; }));
489  // check that the user requests the same number of elements as there really is
490  assert(ranges::distance(flat) == ranges::accumulate(shape, 1, std::multiplies<>{}));
491  // calculate the cummulative product of the shape list in reverse order
492  shape |= ranges::action::reverse;
493  ranges::partial_sum(shape, shape, std::multiplies<>{});
494  // the recursive chunks will share a single copy of the shape list (performance)
495  auto shape_ptr = std::make_shared<std::vector<long>>(std::move(shape));
496  return std::move(flat) | detail::reshaped_view_impl_go<N>::impl(shape_ptr);
497  }
