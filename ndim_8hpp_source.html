259 template<long NDims, typename Rng, typename ValT = ndim_type_t<Rng, NDims>>
260 Rng& ndim_resize(Rng& vec,
261  const std::vector<std::vector<long>>& vec_size,
262  ValT val = ValT{})
263 {
264  // check that the size is valid
265  assert(vec_size.size() == NDims);
266  static_assert(NDims <= ndims<Rng>{} - ndims<ValT>{});
267  for (std::size_t i = 1; i < vec_size.size(); ++i) {
268  assert(vec_size[i].size() == ranges::accumulate(vec_size[i-1], 0UL));
269  }
270  // build initial indices
271  std::vector<long> vec_size_idx(vec_size.size());
272  // recursively resize
273  detail::ndim_resize_impl<1, NDims>::impl(vec, vec_size, vec_size_idx, val);
274  return vec;
275 }

306 template<long NDims, typename Rng, typename ValT = ndim_type_t<Rng, NDims>>
307 Rng& ndim_pad(Rng& vec, ValT val = ValT{})
308 {
309  static_assert(NDims <= ndims<Rng>{} - ndims<ValT>{});
310  std::vector<std::vector<long>> vec_size = ndim_size<NDims>(vec);
311  // replace the sizes in each dimension with the max size in the same dimension
312  for (std::size_t i = 1; i < vec_size.size(); ++i) {
313  long max_size = ranges::max(vec_size[i]);
314  ranges::fill(vec_size[i], max_size);
315  }
316  return utility::ndim_resize<NDims>(vec, vec_size, std::move(val));
317 }

473  template<long N>
474  struct reshaped_view_impl_go {
475  static auto impl(const std::shared_ptr<std::vector<long>>& shape_ptr)
476  {
477  return ranges::views::chunk((*shape_ptr)[N-2])
478  | ranges::views::transform([shape_ptr](auto subview) {
479  return reshaped_view_impl_go<N-1>::impl(shape_ptr)(std::move(subview));
480  });
481  }
482  };
483 
484  template<>
485  struct reshaped_view_impl_go<1> {
486  static auto impl(const std::shared_ptr<std::vector<long>>&)
487  {
488  return ranges::views::all;
489  }
490  };
491 
492  template<long N, typename Rng>
493  auto reshaped_view_impl(Rng& vec, std::vector<long> shape)
494  {
495  assert(shape.size() == N);
496  auto flat = flat_view(vec);
497 
498  // if -1 present in the shape list, deduce the dimension
499  auto deduced_pos = ranges::find(shape, -1);
500  if (deduced_pos != shape.end()) {
501  auto flat_size = ranges::distance(flat);
502  auto shape_prod = -ranges::accumulate(shape, 1, std::multiplies<>{});
503  assert(flat_size % shape_prod == 0);
504  *deduced_pos = flat_size / shape_prod;
505  }
506 
507  // check that all the requested dimenstions have positive size
508  assert(ranges::all_of(shape, [](long s) { return s > 0; }));
509  // check that the user requests the same number of elements as there really is
510  assert(ranges::distance(flat) == ranges::accumulate(shape, 1, std::multiplies<>{}));
511  // calculate the cummulative product of the shape list in reverse order
512  shape |= ranges::actions::reverse;
513  ranges::partial_sum(shape, shape, std::multiplies<>{});
514  // the recursive chunks will share a single copy of the shape list (performance)
515  auto shape_ptr = std::make_shared<std::vector<long>>(std::move(shape));
516  return detail::reshaped_view_impl_go<N>::impl(shape_ptr)(std::move(flat));
517  }
