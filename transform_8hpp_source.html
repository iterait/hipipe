34  // Implementation of partial_transform.
35  template<typename Fun, typename From, typename To>
36  struct partial_transform_impl;

38  template<typename Fun, typename... FromTypes, typename... ToTypes>
39  struct partial_transform_impl<Fun, from_t<FromTypes...>, to_t<ToTypes...>> {
40  Fun fun;
41 
42  batch_t operator()(batch_t source)
43  {
44  // build the view of the selected source columns for the transformer
45  std::tuple<typename FromTypes::data_type&...> slice_view{
46  source.extract<FromTypes>()...
47  };
48  // process the transformer's result and convert it to the requested types
49  static_assert(std::is_invocable_v<Fun&, decltype(slice_view)&&>,
50  "hipipe::stream::partial_transform: "
51  "Cannot apply the given function to the given `from<>` columns.");
52  static_assert(std::is_invocable_r_v<
53  std::tuple<typename ToTypes::data_type...>, Fun&, decltype(slice_view)&&>,
54  "hipipe::stream::partial_transform: "
55  "The function return type does not correspond to the tuple of the "
56  "selected `to<>` columns.");
57  std::tuple<typename ToTypes::data_type...> result =
58  std::invoke(fun, std::move(slice_view));
59  // convert the function results to the corresponding column(s)
60  utility::times_with_index<sizeof...(ToTypes)>([&source, &result](auto i) {
61  using Column = std::tuple_element_t<i, std::tuple<ToTypes...>>;
62  source.insert_or_assign<Column>(std::move(std::get<i>(result)));
63  });
64  return source;
65  }
66  };

124  // Apply fun to each element in tuple of ranges in the given dimension.
125  template<typename Fun, std::size_t Dim, typename From, typename To>
126  struct wrap_fun_for_dim;
127 
128  template<typename Fun, std::size_t Dim, typename... FromTypes, typename... ToTypes>
129  struct wrap_fun_for_dim<Fun, Dim, from_t<FromTypes...>, to_t<ToTypes...>> {
130  Fun fun;
131  using FunRef = decltype(std::ref(fun));
132 
133  utility::maybe_tuple<ToTypes...>
134  operator()(std::tuple<FromTypes&...> tuple_of_ranges)
135  {
136  assert(utility::same_size(tuple_of_ranges));
137  // build the function to be applied
138  wrap_fun_for_dim<FunRef, Dim-1,
139  from_t<ranges::range_value_t<FromTypes>...>,
140  to_t<ranges::range_value_t<ToTypes>...>>
141  fun_wrapper{std::ref(fun)};
142  // transform
143  auto trans_view_of_tuples =
144  ranges::views::transform(
145  std::apply(ranges::views::zip, std::move(tuple_of_ranges)),
146  std::move(fun_wrapper));
147  // unzip the result and convert the ranges to the desired types
148  if constexpr (sizeof...(ToTypes) > 1) {
149  auto trans_tuple_of_vectors =
150  utility::unzip(std::move(trans_view_of_tuples));
151  return convert_tuple_of_ranges<std::tuple<ToTypes...>>(
152  std::move(trans_tuple_of_vectors));
153  // result is only one range, no unzipping
154  } else {
155  return ranges::to<ToTypes...>(std::move(trans_view_of_tuples));
156  }
157  }
158  };

210 auto transform(
211  from_t<FromColumns...> f,
212  to_t<ToColumns...> t,
213  Fun fun,
214  dim_t<Dim> d = dim_t<1>{})
215 {
216  static_assert(
217  ((utility::ndims<typename FromColumns::data_type>::value >= Dim) && ...) &&
218  ((utility::ndims<typename ToColumns::data_type>::value >= Dim) && ...),
219  "hipipe::stream::transform: The dimension in which to apply the operation needs"
220  " to be at most the lowest dimension of all the from<> and to<> columns.");
221 
222  // a bit of function type erasure to speed up compilation
223  using FunT = std::function<
224  utility::maybe_tuple<utility::ndim_type_t<typename ToColumns::data_type, Dim>...>
225  (utility::ndim_type_t<typename FromColumns::data_type, Dim>&...)>;
226  // wrap the function to be applied in the appropriate dimension
227  detail::wrap_fun_for_dim<
228  FunT, Dim,
229  from_t<typename FromColumns::data_type...>,
230  to_t<typename ToColumns::data_type...>>
231  fun_wrapper{std::move(fun)};
232 
233  return stream::partial_transform(f, t, std::move(fun_wrapper));
234 }

240  // wrap the function to be applied only on if the first argument evaluates to true
241  template<typename Fun, typename FromIdxs, typename ToIdxs, typename From, typename To>
242  struct wrap_fun_with_cond;
243 
244  template<typename Fun, std::size_t... FromIdxs, std::size_t... ToIdxs,
245  typename CondCol, typename... Cols, typename... ToTypes>
246  struct wrap_fun_with_cond<Fun,
247  std::index_sequence<FromIdxs...>,
248  std::index_sequence<ToIdxs...>,
249  from_t<CondCol, Cols...>, to_t<ToTypes...>> {
250  Fun fun;
251 
252  utility::maybe_tuple<ToTypes...> operator()(CondCol& cond, Cols&... cols)
253  {
254  // make a tuple of all arguments, except for the condition
255  std::tuple<Cols&...> args_view{cols...};
256  // apply the function if the condition is true
257  if (cond) {
258  // the function is applied only on a subset of the arguments
259  // representing FromColumns
260  static_assert(std::is_invocable_v<Fun&,
261  std::tuple_element_t<FromIdxs, decltype(args_view)>...>,
262  "hipipe::stream::conditional_transform: "
263  "Cannot apply the given function to the given `from<>` columns.");
264  static_assert(std::is_invocable_r_v<
265  std::tuple<ToTypes...>, Fun&,
266  std::tuple_element_t<FromIdxs, decltype(args_view)>...>,
267  "hipipe::stream::conditional_transform: "
268  "The function return type does not correspond to the selected `to<>` columns.");
269  return std::invoke(fun, std::get<FromIdxs>(args_view)...);
270  }
271  // return the original arguments if the condition is false
272  // only a subset of the arguments representing ToColumns is returned
273  // note: We can force std::move in here, because
274  // we are only copying data to themselves.
275  return {std::move(std::get<ToIdxs>(args_view))...};
276  }
277  };

387  template<typename Fun, typename Prng,
388  std::size_t... FromIdxs, std::size_t... ToIdxs,
389  typename... FromTypes, typename... ToTypes>
390  struct wrap_fun_with_prob<Fun, Prng,
391  std::index_sequence<FromIdxs...>,
392  std::index_sequence<ToIdxs...>,
393  from_t<FromTypes...>, to_t<ToTypes...>> {
394  Fun fun;
395  std::reference_wrapper<Prng> prng;
396  const double prob;
397 
398  utility::maybe_tuple<ToTypes...> operator()(FromTypes&... cols)
399  {
400  assert(prob >= 0. && prob <= 1.);
401  std::uniform_real_distribution<> dis{0, 1};
402  // make a tuple of all arguments
403  std::tuple<FromTypes&...> args_view{cols...};
404  // apply the function if the dice roll succeeds
405  if (prob == 1. || (prob > 0. && dis(prng.get()) < prob)) {
406  // the function is applied only on a subset of the arguments
407  // representing FromColumns
408  static_assert(std::is_invocable_v<Fun&,
409  std::tuple_element_t<FromIdxs, decltype(args_view)>...>,
410  "hipipe::stream::probabilistic_transform: "
411  "Cannot apply the given function to the given `from<>` columns.");
412  static_assert(std::is_invocable_r_v<
413  std::tuple<ToTypes...>, Fun&,
414  std::tuple_element_t<FromIdxs, decltype(args_view)>...>,
415  "hipipe::stream::probabilistic_transform: "
416  "The function return type does not correspond to the selected `to<>` columns.");
417  return std::invoke(fun, std::get<FromIdxs>(args_view)...);
418  }
