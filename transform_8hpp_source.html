37  template<typename Fun, typename Projection, typename... SourceTypes,
38  typename... FromTypes, typename... ToTypes>
39  struct partial_transformer<Fun, Projection, std::tuple<SourceTypes...>,
40  from_t<FromTypes...>, to_t<ToTypes...>> {
41  Fun fun;
42  Projection proj;
43 
44  constexpr auto operator()(std::tuple<SourceTypes...> source)
45  {
46  // build the view for the transformer, i.e., slice and project
47  auto slice_view =
48  utility::tuple_transform(utility::tuple_type_view<FromTypes...>(source), proj);
49  // process the transformer's result and convert it to the requested types
50  std::tuple<ToTypes...> result{std::invoke(fun, std::move(slice_view))};
51  // replace the corresponding fields
52  return utility::tuple_cat_unique(std::move(result), std::move(source));
53  }
54  };

114  // Apply fun to each element in tuple of ranges in the given dimension.
115  template<typename Fun, std::size_t Dim, std::size_t NOuts, typename From, typename To>
116  struct wrap_fun_for_dim;

118  template<typename Fun, std::size_t Dim, std::size_t NOuts,
119  typename... FromTypes, typename... ToTypes>
120  struct wrap_fun_for_dim<Fun, Dim, NOuts, from_t<FromTypes...>, to_t<ToTypes...>> {
121  Fun fun;
122  using FunRef = decltype(std::ref(fun));
123 
124  constexpr utility::maybe_tuple<ToTypes...>
125  operator()(std::tuple<FromTypes&...> tuple_of_ranges)
126  {
127  assert(utility::same_size(tuple_of_ranges));
128  // build the function to be applied
129  wrap_fun_for_dim<FunRef, Dim-1, NOuts,
130  from_t<ranges::range_value_type_t<FromTypes>...>,
131  to_t<ranges::range_value_type_t<ToTypes>...>>
132  fun_wrapper{std::ref(fun)};
133  // transform
134  auto range_of_tuples =
135  ranges::view::transform(
136  boost::hana::unpack(std::move(tuple_of_ranges), ranges::view::zip),
137  std::move(fun_wrapper));
138  return utility::unzip_if<(NOuts > 1)>(std::move(range_of_tuples));
139  }
140  };

177 template<typename... FromColumns, typename... ToColumns, typename Fun, int Dim = 1>
178 constexpr auto transform(from_t<FromColumns...> f,
179  to_t<ToColumns...> t,
180  Fun fun,
181  dim_t<Dim> d = dim_t<1>{})

183  // wrap the function to be applied in the appropriate dimension

198  // wrap the function to be applied only on if the first argument evaluates to true
199  template<typename Fun, typename FromIdxs, typename ToIdxs, typename From, typename To>
200  struct wrap_fun_with_cond;
201 
202  template<typename Fun, typename FromIdxs, std::size_t... ToIdxs,
203  typename CondCol, typename... Cols, typename... ToTypes>
204  struct wrap_fun_with_cond<Fun, FromIdxs, std::index_sequence<ToIdxs...>,
205  from_t<CondCol, Cols...>, to_t<ToTypes...>> {
206  Fun fun;
207 
208  constexpr utility::maybe_tuple<ToTypes...> operator()(CondCol& cond, Cols&... cols)
209  {
210  // make a tuple of all arguments, except for the condition
211  std::tuple<Cols&...> args_view{cols...};
212  // apply the function if the condition is true
213  if (cond) {
214  // the function is applied only on a subset of the arguments
215  // representing FromColumns
216  return boost::hana::unpack(
217  utility::tuple_index_view(args_view, FromIdxs{}), fun);
218  }
219  // return the original arguments if the condition is false
220  // only a subset of the arguments representing ToColumns is returned
221  // note: We can force std::move in here, because
222  // we are only copying data to themselves.
223  return {std::move(std::get<ToIdxs>(args_view))...};
224  }
225  };

318  // wrap the function to be an identity if the dice roll fails
319  template<typename Fun, typename Prng,
320  typename FromIdxs, typename ToIdxs,
321  typename From, typename To>
322  struct wrap_fun_with_prob;
323 
324  template<typename Fun, typename Prng,
325  typename FromIdxs, std::size_t... ToIdxs,
326  typename... FromTypes, typename... ToTypes>
327  struct wrap_fun_with_prob<Fun, Prng,
328  FromIdxs, std::index_sequence<ToIdxs...>,
329  from_t<FromTypes...>, to_t<ToTypes...>> {
330  Fun fun;
331  std::reference_wrapper<Prng> prng;
332  const double prob;
333 
334  utility::maybe_tuple<ToTypes...> operator()(FromTypes&... cols)
335  {
336  assert(prob >= 0. && prob <= 1.);
337  std::uniform_real_distribution<> dis{0, 1};
338  // make a tuple of all arguments
339  std::tuple<FromTypes&...> args_view{cols...};
340  // apply the function if the dice roll succeeds
341  if (prob == 1. || (prob > 0. && dis(prng.get()) < prob)) {
342  // the function is applied only on a subset of the arguments
343  // representing FromColumns
344  return boost::hana::unpack(
345  utility::tuple_index_view(args_view, FromIdxs{}), fun);
346  }
347  // return the original arguments if the dice roll fails
348  // only a subset of the arguments representing ToColumns is returned
349  // note: We can force std::move in here, because
350  // we are only copying data to themselves.
351  return {std::move(std::get<ToIdxs>(args_view))...};
352  }
353  };
