714  static void throw_check_new_header(
715  std::size_t n_cols,
716  const std::vector<std::string>& header)
717  {
718  if (header.size() && header.size() != n_cols) {
719  throw std::invalid_argument{"The dataframe with " + std::to_string(n_cols) +
720  " columns cannot have a header of size " + std::to_string(header.size()) + "."};
721  }
722  for (const std::string& h : header) {
723  if (!h.size()) {
724  throw std::invalid_argument{"When providing a header to a dataframe,"
725  " all the column names have to be non-empty."};
726  }
727  }
728  }

786  template <typename This>
787  static auto raw_irows_impl(This this_ptr, std::vector<std::size_t> col_indexes)
788  {
789  namespace view = ranges::view;
790  return view::iota(0UL, this_ptr->n_rows())
791  | view::transform([this_ptr, col_indexes=std::move(col_indexes)](std::size_t i) {
792  return this_ptr->raw_icols(col_indexes)
793  // decltype(auto) to make sure a reference is returned
794  | view::transform([i](auto&& col) -> decltype(auto) {
795  return col[i];
796  });

836 template<typename DataTable>
837 std::ostream& operator<<(std::ostream& out, const dataframe<DataTable>& df)
838 {
839  namespace view = ranges::view;
840  // calculate the width of the columns using their longest field
841  std::vector<std::size_t> col_widths = df.raw_cols()
842  | view::transform([](auto&& col) {
843  std::vector<std::size_t> elem_sizes = col
844  | view::transform([](auto& field) { return ranges::size(field); });
845  return ranges::max(elem_sizes) + 2;
846  });
847 
848  auto header = df.header();
849  if (header.size()) {
850  // update col_widths using header widths
851  col_widths = view::zip(col_widths, header)
852  | view::transform([](auto&& tpl) {
853  return std::max(std::get<0>(tpl), std::get<1>(tpl).size() + 2);
854  });
855 
856  // print header
857  for (std::size_t j = 0; j < header.size(); ++j) {
858  out << std::setw(col_widths[j]) << header[j];
859  if (j + 1 < header.size()) out << '|';
860  else out << '\n';
861  }
862 
863  // print header and data separator
864  for (std::size_t j = 0; j < header.size(); ++j) {
865  out << std::setw(col_widths[j]) << std::setfill('-');
866  if (j + 1 < header.size()) out << '-' << '+';
867  else out << '-' << '\n';
868  }
869  out << std::setfill(' ');
870  }
871 
872  // print data
873  for (std::size_t i = 0; i < df.n_rows(); ++i) {
874  for (std::size_t j = 0; j < df.n_cols(); ++j) {
875  out << std::setw(col_widths[j]) << df.raw_rows()[i][j];
876  if (j + 1 < df.n_cols()) out << '|';
877  else out << '\n';
878  }
879  }
880 
881  return out;
882 }
