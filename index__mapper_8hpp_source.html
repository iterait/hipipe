110  std::size_t insert(T val)
111  {
112  if (contains(val)) {
113  throw std::invalid_argument{"The element is already present in the index mapper."};
114  }
115  val2idx_[val] = idx2val_.size();
116  idx2val_.push_back(std::move(val));
117  return idx2val_.size() - 1;
