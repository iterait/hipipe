20  // Create a transformation function that can be sent to stream::transform.
21  template<typename FromColumn, typename ToColumn, typename Gen, int Dim>
22  struct wrap_generate_fun_for_transform {
23  Gen gen;
24  long gendims;
25 
26  typename ToColumn::batch_type operator()(typename FromColumn::batch_type& source)
27  {
28  using SourceVector = typename FromColumn::batch_type;
29  using TargetVector = typename ToColumn::batch_type;
30  constexpr long SourceDims = utility::ndims<SourceVector>::value;
31  static_assert(Dim <= SourceDims, "stream::generate requires"
32  " the dimension in which to apply the generator to be at most the number"
33  " of dimensions of the source column (i.e., the column the shape is taken"
34  " from).");
35  // get the size of the source up to the dimension of the target
36  std::vector<std::vector<long>> target_size = utility::ndim_size<Dim>(source);
37  // create, resize, and fill the target using the generator
38  TargetVector target;
39  utility::ndim_resize<Dim>(target, target_size);
40  utility::generate<Dim>(target, gen, gendims);
41  return target;
42  }
