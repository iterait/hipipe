32 template<typename Container2d>
33 Container2d transpose(const Container2d& data)
34 {
35  auto width = ranges::size(data);
36  auto height = ranges::size(ranges::at(data, 0));
37  Container2d res(height, ranges::range_value_type_t<Container2d>(width));
38  for (std::size_t i = 0; i < height; ++i) {
39  for (std::size_t j = 0; j < width; ++j) {
40  res[i][j] = data[j][i];
41  }
42  }
43  return res;
44 }
