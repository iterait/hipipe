29 auto generate_batched_data(std::vector<int> batch_sizes)
30 {
31  std::vector<std::tuple<Unique, Shared>> data;
32  int counter = 0;
33  for (std::size_t i = 0; i < batch_sizes.size(); ++i) {
34  std::vector<std::unique_ptr<int>> unique_data;
35  std::vector<std::shared_ptr<int>> shared_data;
36  for (int j = 0; j < batch_sizes[i]; ++j) {
37  unique_data.emplace_back(std::make_unique<int>(counter));
38  shared_data.emplace_back(std::make_shared<int>(counter));
39  ++counter;
40  }
41  data.emplace_back(std::make_tuple(std::move(unique_data), std::move(shared_data)));
42  }
43  return data;
44 }

51 template<typename Data>
52 void check_20_elems_batch_size_2(Data data)
53 {
54  auto rng = data | ranges::view::move | batch(3);
55  using tuple_type = decltype(*ranges::begin(rng));
56  static_assert(std::is_same<std::tuple<Unique, Shared>&, tuple_type>{});
57  using first_type = decltype(std::get<0>(*ranges::begin(rng)).value()[0]);
58  static_assert(std::is_same<std::unique_ptr<int>&, first_type>{});
59  using second_type = decltype(std::get<1>(*ranges::begin(rng)).value()[0]);
60  static_assert(std::is_same<std::shared_ptr<int>&, second_type>{});
61 
62  // iterate through batches
63  std::vector<int> result_unique;
64  std::vector<int> result_shared;
65  int batch_n = 0;
66  int n = 0;
67  for (auto&& tuple : rng) {
68  // the last batch should be smaller
69  if (batch_n == 6) {
70  BOOST_TEST(std::get<0>(tuple).value().size() == 2U);
71  BOOST_TEST(std::get<1>(tuple).value().size() == 2U);
72  }
73  else {
74  BOOST_TEST(std::get<0>(tuple).value().size() == 3U);
75  BOOST_TEST(std::get<1>(tuple).value().size() == 3U);
76  }
77  BOOST_CHECK(is_same_batch_size(tuple));
78 
79  // iterate through batch values
80  for (auto& elem : std::get<0>(tuple).value()) {
81  result_unique.push_back(*elem);
82  ++n;
83  }
84  for (auto& elem : std::get<1>(tuple).value()) {
85  result_shared.push_back(*elem);
86  }
87  ++batch_n;
88  }
89  BOOST_TEST(batch_n == 7);
90  BOOST_TEST(n == 20);
91 
92  auto desired = ranges::view::iota(0, 20);
93  test_ranges_equal(result_unique, desired);
94  test_ranges_equal(result_shared, desired);
95 }
