53  data.emplace_back(1, std::make_unique<int>(2));

55  auto generated = data
56  | ranges::view::move
57  | transform(from<Unique>, to<Unique, Double>,
58  [](const std::unique_ptr<int> &ptr) {

60  })
61  | ranges::to_vector;
62
